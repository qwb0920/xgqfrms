Hex RGB

				Numerical color values
				Hex rgb	Decimal

				em { color: #f00 }              /* #rgb */
				em { color: #ff0000 }           /* #rrggbb */
				em { color: rgb(255,0,0) }
				em { color: rgb(100%, 0%, 0%) }

RGBA

				RGBA color values

				em { color: rgba(100%,0%,0%,1) } /* the same, with explicit opacity of 1 */
				p { color: rgba(100%, 50%, 0%, 0.1) } /* very transparent solid orange */

HSV/HSL

				(RGB/Hex)
				(RGB)   Red + Green + Blue
				(RGB)   (0-255) + (0-255) + (0-255)
				(RGB)   (0%-100%) + (0%-100%) + (0%-100%)
				(Hex)   (00-FF) + (00-FF) + (00-FF)
				
				(HSL/HSV)	
				(HSL)	Hue + Satur + Light
				(HSL)	(0°～360°) + (0%-100%) + (0%-100%)
				(HSV)	Hue + Satur + Value
				(HSV)	(0°～360°) + (0%-100%) + (0%-100%)

				Alias
				
Alpha (0-1)

				从RGB到HSL或HSV的转换
				
				
				
				
				
			
				HSL color values

				* { color: hsl(0, 100%, 50%) }   /* red */
				* { color: hsl(120, 100%, 50%) } /* lime */ 
				* { color: hsl(120, 100%, 25%) } /* dark green */ 
				* { color: hsl(120, 100%, 75%) } /* light green */ 
				* { color: hsl(120, 75%, 75%) }  /* pastel green, and so on */

HSLa

				/* HSL model with alpha channel */

				p { color: hsla(120, 100%, 50%, 1) } /* green */
				p { color: hsla(120, 100%, 50%, 0.5) } /* semi-transparent green */
				p { color: hsla(120, 100%, 50%, 0.1) } /* very transparent green */
